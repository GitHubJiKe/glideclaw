agents (代理/助手)
  ├─ id: 主键
  ├─ name: 名称
  ├─ description: 描述
  └─ created_at: 创建时间

souls (灵魂/系统提示)
  ├─ id: 主键
  ├─ assistant_id: 关联助手
  ├─ version: 版本
  ├─ prompt: 系统提示词
  └─ created_at: 创建时间

identities (身份)
  ├─ id: 主键
  ├─ assistant_id: 关联助手
  ├─ name: 身份名称
  ├─ description: 身份描述
  └─ created_at: 创建时间

heartbeats (心跳/定时任务) ✨ 新增
  ├─ id: 主键
  ├─ assistant_id: 关联助手
  ├─ event_type: 事件类型
  ├─ description: 任务描述
  ├─ cron_expression: Cron 表达式
  ├─ enabled: 是否启用
  ├─ last_run_at: 上次运行时间
  ├─ next_run_at: 下次运行时间
  └─ created_at: 创建时间

users (用户)
  ├─ id: 主键
  ├─ name: 名称
  ├─ email: 邮箱
  └─ created_at: 创建时间

change_history (变更历史)
  ├─ id: 自增主键
  ├─ table_name: 表名
  ├─ row_id: 行 ID
  ├─ operation: 操作类型
  ├─ payload: 数据负载
  └─ timestamp: 时间戳
